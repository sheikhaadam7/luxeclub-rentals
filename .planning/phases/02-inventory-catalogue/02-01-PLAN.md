```sql
CREATE TABLE bookings (
  id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  vehicle_id      UUID REFERENCES vehicles(id) ON DELETE CASCADE,
  user_id         UUID REFERENCES auth.users(id),
  start_date      DATE NOT NULL,
  end_date        DATE NOT NULL,
  status          TEXT NOT NULL DEFAULT 'pending'
    CHECK (status IN ('pending', 'confirmed', 'completed', 'cancelled')),
  created_at      TIMESTAMPTZ NOT NULL DEFAULT NOW()
);
ALTER TABLE bookings ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Users can view own bookings"
  ON bookings FOR SELECT TO authenticated USING (auth.uid() = user_id);
```

Create RPC function for blocked dates (avoids exposing full booking rows):
```sql
CREATE OR REPLACE FUNCTION get_blocked_dates(p_vehicle_id UUID)
RETURNS TABLE(start_date DATE, end_date DATE) AS $$
BEGIN
  RETURN QUERY
  SELECT b.start_date, b.end_date
  FROM bookings b
  WHERE b.vehicle_id = p_vehicle_id
    AND b.status IN ('pending', 'confirmed')
    AND b.end_date >= CURRENT_DATE;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;
```

**4. `scripts/scraper/run.ts`** — Entry point:

```typescript
import { chromium } from 'playwright'
import { createClient } from '@supabase/supabase-js'
import { scrapeVehicles } from './scrape-vehicles'
import { uploadImages } from './upload-images'
import { upsertVehicles } from './upsert-db'

async function run() {
  // Validate env vars: NEXT_PUBLIC_SUPABASE_URL, SUPABASE_SERVICE_ROLE_KEY
  const supabase = createClient(
    process.env.NEXT_PUBLIC_SUPABASE_URL!,
    process.env.SUPABASE_SERVICE_ROLE_KEY!,
  )

  const browser = await chromium.launch({ headless: true })
  const context = await browser.newContext()
  const page = await context.newPage()

  // Block non-data resources for performance
  await page.route('**/*', (route) => {
    const type = route.request().resourceType()
    if (['image', 'stylesheet', 'font', 'media'].includes(type)) {
      route.abort()
    } else {
      route.continue()
    }
  })
  // NOTE: resource blocking is only active during scraping phase.
  // When downloading images in upload-images.ts, use fetch() directly (not page navigation),
  // so the route blocking does not interfere.

  try {
    console.log('Scraping vehicles from luxeclubrentals.com...')
    const vehicles = await scrapeVehicles(page)
    console.log(`Found ${vehicles.length} vehicles`)

    if (vehicles.length === 0) {
      throw new Error('No vehicles found — scraper selectors may be broken')
    }

    // Re-enable image loading for download phase (or use fetch which is unaffected)
    console.log('Uploading images to Supabase Storage...')
    const imageMap = await uploadImages(supabase, vehicles)

    console.log('Upserting vehicle records...')
    const result = await upsertVehicles(supabase, vehicles, imageMap)
    console.log(`Done: ${result.count} vehicles upserted`)
  } catch (err) {
    const errorMsg = err instanceof Error ? err.message : String(err)
    console.error('Scraper failed:', errorMsg)
    // Log failure to scraper_runs
    await supabase.from('scraper_runs').insert({
      status: 'error',
      error_msg: errorMsg,
      vehicle_count: 0,
    })
    process.exit(1)
  } finally {
    await browser.close()
  }
}

run()
```
